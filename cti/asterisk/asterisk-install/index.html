HowTos: Asterisk install

  • 选择版本安装
  • 安装相应依赖
  • make menuselect
  • 官方版本：http://downloads.asterisk.org/pub/telephony/asterisk/releases/ #下载13.9.1
  • pjsip: http://www.pjsip.org/release/2.5.1/pjproject-2.5.1.tar.bz2

保持系统最新

apt-get update && apt-get upgrade -y

解决系统依赖包 gcc..:

apt-get -y install build-essential subversion git

pjproject source

cd /usr/src/
git clone https://github.com/asterisk/pjproject.git
or 
wget http://www.pjsip.org/release/2.5.1/pjproject-2.5.1.tar.bz2
tar -jxvf pjproject-*
cd pjproject
./configure --enable-shared --disable-sound --disable-resample --disable-video --disable-opencore-amr
make dep && make && make install

Download and unpack DAHDI

wget http://downloads.asterisk.org/pub/telephony/dahdi-linux-complete/dahdi-linux-complete-current.tar.gz
tar zxvf dahdi-linux-complete-current.tar.gz
cd dahdi-linux-complete-2.10.0.1+2.10.0.1/
make all
make install
make config

LIBPRI :

wget http://downloads.asterisk.org/pub/telephony/libpri/libpri-1.4-current.tar.gz
tar zxvf libpri-1.4-current.tar.gz
cd libpri-1.4.15/
make
make install

asterisk-current:

wget http://downloads.asterisk.org/pub/telephony/asterisk/asterisk-13-current.tar.gz
tar -zxvf asterisk-13-current.tar.gz
cd asterisk-current*
#Add mp3 support
contrib/scripts/get_mp3_source.sh
ln -s addons/mp3 mp3
#检查依赖：
./contrib/scripts/install_prereq install 
./contrib/scripts/install_prereq install-unpackaged
./bootstrap.sh
./configure --enable-realtime-only --enable-error-concealment --disable-examples --enable-opus --enable-vp8 --enable-pic --enable-shared  
make menuselect #adds-one 的所有项选上,选上format_mp3

#注意用到fax功能的，要选上res_fax。
make && make install 
make config

添加asterisk:asterisk ,并将Asterisk改由asterisk 用户运行：

adduser asterisk --disabled-password --shell /sbin/nologin  --no-create-home -d /var/lib/asterisk --gecos "Asterisk User"
or
useradd -c "Asterisk User" -d /var/lib/asterisk asterisk

vi /etc/default/asterisk
  AST_USER="asterisk"
  AST_GROUP="asterisk"

将asterisk conf asterisk.ctl给与 asterisk组权限

chown asterisk:asterisk /etc/asterisk -R
chown asterisk:asterisk /var/run/asterisk -R
chown -R asterisk. /var/lib/asterisk

或者制作 deb包:

apt-get build-dep asterisk 安装所有编译asterisk依赖
debuild -i -us -uc -b

Find build dependencies, don't need this because we don't care about the source package.

dpkg-depcheck -d ./configure

# Build only binary packages and don't sign them
dpkg-buildpackage -us -uc -c -j8

附： 生成证书，如webRTC TLS,需要：

cd /usr/local/src/asterisk-11.13.1~dfsg/contrib/scripts
./ast_tls_cert -C pbx.mycompany.com -O "My Super Company" -d /etc/asterisk/keys  
./ast_tls_cert -m client -c /etc/asterisk/keys/ca.crt -k /etc/asterisk/keys/ca.key -C 192.168.10.3 -O "My Super Company" -d /etc/asterisk/keys -o malcolm

opus vp8 patch:

  • https://github.com/netaskd/asterisk-opus/ #11.5.0
  • https://github.com/seanbright/asterisk-opus #13.3.2
  • https://github.com/traud/asterisk-silk #13.7.1

cp ../asterisk-opus/formats/* formats/
patch -p1 < ../asterisk-opus/asterisk.patch

apt-get install opus-tool
